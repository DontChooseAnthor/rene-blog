function Promise(executor){
    var self = this
    self.state = 'padding'
    self.value = ''
    self.reason = ''
    self.onResolved = []
    self.onRejected = []
    function resolve(value){
        if(self.state === 'padding'){
        self.state = 'resolved'
        self.value = value
        self.onResolved.forEach()
      }
   }
    function reject(reason){
        if(self.state === 'padding'){
        self.state = 'rejected'
        self.reason = reason
        self.onRejected.forEach()
      }
   }
    try{
        executor(resolve,reject)
    }catch(e){
        reject(e)
    }
}
